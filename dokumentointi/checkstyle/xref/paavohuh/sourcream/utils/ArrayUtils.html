4   import org.joou.UShort;

6   // Due to Java's lackluster generics (no generic instantation, no support for
7   // value types as generic arguments), this class is bit of a mess. Not much can
8   // be done about it.
9   
10  /**
11   * Contains utilities for manipulating arrays.
12   */
13  public class ArrayUtils {   
14      public static byte[] clone(byte[] array) {
15          byte[] clone = new byte[array.length];
16          System.arraycopy(array, 0, clone, 0, array.length);
17          return clone;
18      }
19      
20      public static UByte[] clone(UByte[] array) {
21          UByte[] clone = new UByte[array.length];
22          System.arraycopy(array, 0, clone, 0, array.length);
23          return clone;
24      }
25      
26      public static UShort[] clone(UShort[] array) {
27          UShort[] clone = new UShort[array.length];
28          System.arraycopy(array, 0, clone, 0, array.length);
29          return clone;
30      }
31      
32      public static boolean[][] clone(boolean[][] array) {
33          boolean[][] clone = new boolean[array.length][];
34          
35          for (int y = 0; y < array.length; y++) {
36              clone[y] = array[y].clone();
37          }
38          
39          return clone;
40      }
41      
42      public static <T> T[] clone(T[] array) {
43          Object[] clone = new Object[array.length];
44          System.arraycopy(array, 0, clone, 0, array.length);
45          return (T[]) clone;
46      }
47  }
