2   package paavohuh.sourcream.utils;

4   import java.util.function.Supplier;
5   
6   /**
7    * A lazy wrapper for an object. The value is only initialized once and only
8    * when required.
9    * @param <T> The type of the stored object
10   */
11  public class Cached<T> {
12      private T value;
13      private final Supplier<T> initializer;
14  
15      /**
16       * Creates a new Cached instace. The value is initialized with the
17       * initializer when get() is called.
18       * @param initializer 
19       */
20      public Cached(Supplier<T> initializer) {
21          this.initializer = initializer;
22      }
23      
24      /**
25       * Gets the value. If it doesn't exist yet, the initializer is run and the
26       * value is returned.
27       * @return The value
28       */
29      public T get() {
30          if (value == null) {
31              value = initializer.get();
32          }
33          
34          return value;
35      }
36  }
