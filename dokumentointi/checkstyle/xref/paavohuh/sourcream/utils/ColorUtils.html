4   import java.awt.Color;

6   public class ColorUtils {
7       
8       public static Color lerp(Color one, Color zero, float level) {
9           float inverseLevel = 1.0f - level;
10                  
11          float red =
12              (one.getRed() / 255f) * level +
13              (zero.getRed() / 255f) * inverseLevel;
14  
15          float green =
16              (one.getGreen() / 255f) * level +
17              (zero.getGreen() / 255f) * inverseLevel;
18  
19          float blue =
20              (one.getBlue() / 255f) * level +
21              (zero.getBlue() / 255f) * inverseLevel;
22  
23          return new Color(
24              MathUtils.clamp(0.0f, red, 1.0f),
25              MathUtils.clamp(0.0f, green, 1.0f),
26              MathUtils.clamp(0.0f, blue, 1.0f));
27      }
28  }
