4   import java.util.Optional;
5   
6   /**
7    * Contains utility functions for working with Optional objects.
8    */
9   public class OptionalUtils {
10      /**
11       * Takes a builder function, runs it and wrap the result into an Optional.
12       * If the builder throws an exception, returns an empty Optional.
13       * @param <T> The type stored in Optional
14       * @param builder A function that might throw and returns an instance of T.
15       * @return The result of the function call, or an empty Optional.
16       */
17      public static <T> Optional<T> ofThrowing(ThrowingSupplier<T> builder) {
18          try {
19              return Optional.of(builder.get());
20          } catch (Exception e) {
21              return Optional.empty();
22          }
23      }
24      
25      @FunctionalInterface
26      /**
27       * A supplier that can throw exceptions.
28       * @param <T> The value returned by the supplier.
29       */
30      public interface ThrowingSupplier<T> {
31          /**
32           * Supplies a value of type T.
33           * @return A value of type T.
34           * @throws Exception Can throw any exception.
35           */
36          T get() throws Exception;
37      }
38  }
