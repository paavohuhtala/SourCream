2   package paavohuh.sourcream.ui;
3   
4   import java.awt.Color;
5   import java.awt.Dialog;
6   import java.awt.Dimension;
7   import java.awt.FontMetrics;
8   import java.awt.Graphics;
9   import java.util.function.Consumer;
10  import javax.swing.JButton;
11  import paavohuh.sourcream.utils.ColorUtils;
12  
13  /**
14   * A control for selecting colors. Creates a ColorChooserDialog on keypress.
15   */
16  public final class ColorChooserButton extends JButton {
17      private Color color;
18  
19      /**
20       * Creates a new color chooser button.
21       * @param parent The parent.
22       * @param label The label of the button.
23       * @param initialColor The initial color.
24       */
25      public ColorChooserButton(Dialog parent, String label, Color initialColor) {
26          //this.setPreferredSize(new Dimension(64, 64));
27          this.setMinimumSize(new Dimension(64, 32));
28          this.setOpaque(true);
29          this.setBorderPainted(false);
30          this.setText(label);
31  
32          setColor(initialColor);
33          
34          this.addActionListener(event -> {
35              ColorChooserDialog chooser = new ColorChooserDialog(parent, color);
36              setColor(chooser.showAsDialog());
37          });
38      }
39  
40      public Color getColor() {
41          return color;
42      }
43  
44      /**
45       * Sets the color.
46       * @param color The color.
47       */
48      public void setColor(Color color) {
49          this.color = color;
50          this.setBackground(color);
51          this.setForeground(ColorUtils.invert(color));
52      }
53      
54      /**
55       * Register a new callback for checking when the color is changed.
56       * @param func 
57       */
58      public void onChangeColor(Consumer<Color> func) {
59          addChangeListener(event -> {
60              func.accept(getColor());
61          });
62      }
63      
64      @Override
65      public void paintComponent(Graphics g) {
66          super.paintComponent(g);
67          g.setColor(color);
68          g.fillRect(2, 2, getWidth() - 4, getHeight() - 4);
69          g.setColor(getForeground());
70          
71          FontMetrics metrics = g.getFontMetrics();
72          int textHeight = metrics.getHeight();
73          int textWidth = metrics.stringWidth(getText());
74          
75          g.drawString(getText(), getWidth() / 2 - textWidth / 2, getHeight() / 2 + textHeight / 2);
76      }
77  }
