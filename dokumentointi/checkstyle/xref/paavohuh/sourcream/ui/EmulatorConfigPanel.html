5   import java.awt.*;
6   import javax.swing.*;
7   import paavohuh.sourcream.configuration.EmulatorConfiguration;
8   import paavohuh.sourcream.configuration.KnownBindings;
9   
10  /**
11   * The emulator config panel.
12   */
13  public class EmulatorConfigPanel extends JPanel {
14  
15      private final EmulatorConfiguration config;
16      
17      /**
18       * Creates a new emulator config panel.
19       * @param parent The parent. 
20       * @param config The emulator configuration.
21       */
22      public EmulatorConfigPanel(Dialog parent, EmulatorConfiguration config) {
23          this.config = config;
24          
25          setLayout(new GridLayout(1, 2, 4, 4));
26          
27          JPanel leftColumn = new JPanel(new GridLayout(4, 1));
28          leftColumn.setBorder(BorderFactory.createTitledBorder("Screen"));
29          JPanel rightColumn = new JPanel(new BorderLayout());
30          rightColumn.setBorder(BorderFactory.createTitledBorder("Input"));
31          
32          JPanel colorGroup = new JPanel(new GridLayout(1, 2, 15, 15));
33          colorGroup.setBorder(BorderFactory.createTitledBorder("Colors"));
34          
35          ColorChooserButton bg = new ColorChooserButton(parent, "Background", config.getScreen().getColors().getBackground());
36          ColorChooserButton fg = new ColorChooserButton(parent, "Foreground", config.getScreen().getColors().getForeground());
37          
38          bg.onChangeColor(color -> {
39              config.getScreen().getColors().setBackground(color);
40          });
41          
42          fg.onChangeColor(color -> {
43              config.getScreen().getColors().setForeground(color);
44          });
45          
46          colorGroup.add(bg);
47          colorGroup.add(fg);
48          leftColumn.add(colorGroup);
49          
50          JPanel bindingsGroup = new JPanel(new BorderLayout(4, 4));
51          JComboBox presetBox = new JComboBox(new String[]{"Custom", "Pong", "Tetris"});
52          
53          // Temporary HACK
54          presetBox.addActionListener(event -> {
55              String selected = (String) presetBox.getModel().getSelectedItem();
56              
57              System.out.println(selected);
58              switch (selected) {
59                  case "Pong":
60                      config.setInput(KnownBindings.pong());
61                      break;
62                  case "Tetris":
63                      config.setInput(KnownBindings.tetris());
64                      break;
65              }
66          });
67          
68          bindingsGroup.add(presetBox, BorderLayout.NORTH);
69          
70          JPanel subBindingsGroup = new JPanel(new GridLayout(8, 2, 4, 4));
71          
72          String[] keys = {"None", "Up", "Down", "Left", "Right", "W", "A", "S", "D"};
73          
74          for (int i = 0; i < 16; i++) {
75              JPanel bindingPanel = new JPanel(new BorderLayout(4, 0));
76              bindingPanel.add(new JLabel(Integer.toString(i + 1)), BorderLayout.WEST);
77              JComboBox keyBox = new JComboBox(keys);
78              keyBox.setPreferredSize(new Dimension(80, 0));
79              bindingPanel.add(keyBox, BorderLayout.EAST);
80              subBindingsGroup.add(bindingPanel);
81          }
82          
83          bindingsGroup.add(subBindingsGroup, BorderLayout.CENTER);
84          
85          rightColumn.add(bindingsGroup, BorderLayout.CENTER);
86          
87          JPanel ghostingGroup = new JPanel(new GridLayout(3, 2, 4, 4));
88          ghostingGroup.setBorder(BorderFactory.createTitledBorder("Ghosting"));
89          ghostingGroup.add(new JCheckBox("Enable?"));
90          ghostingGroup.add(new JLabel());
91          
92          SpinnerModel fadeInModel = new SpinnerNumberModel(
93              config.getScreen().getGhosting().getAddBy(), 0.1, 1.0, 0.1);
94          JSpinner fadeIn = new JSpinner(fadeInModel);
95          JSpinner fadeOut = new JSpinner();
96          
97          ghostingGroup.add(new JLabel("Fade in"));
98          ghostingGroup.add(fadeIn);
99          ghostingGroup.add(new JLabel("Fade out"));
100         ghostingGroup.add(fadeOut);
101 
102         leftColumn.add(ghostingGroup);
103         
104         JPanel scaleGroup = new JPanel(new GridLayout(1, 2));
105         scaleGroup.add(new JLabel("Scale factor"));
106         scaleGroup.add(new JSpinner(new SpinnerNumberModel(10, 1, 50, 1)));
107         leftColumn.add(scaleGroup);
108                 
109         add(leftColumn);
110         add(rightColumn);
111     }
112 
113     public EmulatorConfiguration getConfig() {
114         return config;
115     }
116 }
