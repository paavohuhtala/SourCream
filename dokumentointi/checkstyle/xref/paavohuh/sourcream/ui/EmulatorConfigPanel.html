30  public class EmulatorConfigPanel extends JPanel {
31  
32      private final EmulatorConfiguration config;
33      private final List<JComboBox<NamedKey>> keyBindBoxes;
34      private final Integer[] keyBindings;
35      
36      /**
37       * Creates a new emulator config panel.
38       * @param parent The parent. 
39       * @param config The emulator configuration.
40       */
41      public EmulatorConfigPanel(Dialog parent, EmulatorConfiguration config) {
42          this.config = config;
43          
44          setLayout(new GridLayout(1, 2, 4, 4));
45          
46          GridBagConstraints c = new GridBagConstraints();
47          c.fill = GridBagConstraints.BOTH;
48          c.anchor = GridBagConstraints.PAGE_START;
49          c.weightx = 1.0f;
50          
51          JPanel leftColumn = new JPanel(new GridBagLayout());
52          JPanel screenGroup = new JPanel(new GridBagLayout());
53          
54          screenGroup.setBorder(BorderFactory.createTitledBorder("Screen"));
55          
56          JPanel rightColumn = new JPanel(new BorderLayout());
57          rightColumn.setBorder(BorderFactory.createTitledBorder("Input"));
58          
59          JPanel colorGroup = new JPanel(new GridLayout(1, 2, 15, 15));
60          colorGroup.setBorder(BorderFactory.createTitledBorder("Colors"));
61          
62          ColorChooserButton bg = new ColorChooserButton(parent, "Background", config.getScreen().getColors().getBackground());
63          ColorChooserButton fg = new ColorChooserButton(parent, "Foreground", config.getScreen().getColors().getForeground());
64          
65          bg.onChangeColor(color -> {
66              config.getScreen().getColors().setBackground(color);
67          });
68          
69          fg.onChangeColor(color -> {
70              config.getScreen().getColors().setForeground(color);
71          });
72          
73          colorGroup.add(bg);
74          colorGroup.add(fg);
75          
76          c.weighty = 0.7;
77          c.gridx = 0;
78          c.gridy = 0;
79          c.gridwidth = 1;
80          c.gridheight = 2;
81          
82          screenGroup.add(colorGroup, c);
83          
84          JPanel bindingsGroup = new JPanel(new BorderLayout(4, 4));
85          JComboBox presetBox = new JComboBox(new String[]{"Custom", "Pong", "Tetris", "UFO", "Tank"});
86          
87          presetBox.addActionListener(event -> {
88              String selected = (String) presetBox.getModel().getSelectedItem();
89              
90              switch (selected) {
91                  case "Pong":
92                      updateConfigBindings(KnownBindings.pong());
93                      break;
94                  case "Tetris":
95                      updateConfigBindings(KnownBindings.tetris());
96                      break;
97                  case "UFO":
98                      updateConfigBindings(KnownBindings.ufo());
99                      break;
100                 case "Tank":
101                     updateConfigBindings(KnownBindings.tank());
102                     break;
103             }
104             
105             updateDisplayedBindings();
106         });
107         
108         bindingsGroup.add(presetBox, BorderLayout.NORTH);
109         
110         JPanel subBindingsGroup = new JPanel(new GridLayout(8, 2, 4, 4));
111         
112         List<NamedKey> namedKeyList = NamedKey.getKnown();
113         NamedKey[] keys = namedKeyList.toArray(new NamedKey[namedKeyList.size()]);
114 
115         keyBindBoxes = new ArrayList<>(16);
116         keyBindings = new Integer[16];
117         
118         for (int i = 0; i < 16; i++) {
119             final int index = i;
120             
121             JPanel bindingPanel = new JPanel(new BorderLayout(4, 0));
122             bindingPanel.add(new JLabel(Integer.toString(i + 1)), BorderLayout.WEST);
123             
124             JComboBox<NamedKey> keyBindBox = new JComboBox(keys);
125             keyBindBox.setPreferredSize(new Dimension(80, 0));
126             keyBindBox.addItemListener(event -> {
127                 NamedKey key = (NamedKey) event.getItem();
128                 keyBindings[index] = key.getKeyCode();
129                 updateConfigBindings();
130             });
131             
132             bindingPanel.add(keyBindBox, BorderLayout.EAST);
133             subBindingsGroup.add(bindingPanel);
134             keyBindBoxes.add(keyBindBox);
135         }
136         
137         updateDisplayedBindings();
138         
139         bindingsGroup.add(subBindingsGroup, BorderLayout.CENTER);
140         rightColumn.add(bindingsGroup, BorderLayout.CENTER);
141         
142         JPanel ghostingGroup = new JPanel(new GridLayout(3, 2, 4, 4));
143         ghostingGroup.setBorder(BorderFactory.createTitledBorder("Ghosting"));
144         
145         JCheckBox enableGhosting = new JCheckBox("Enable?");
146         
147         enableGhosting.setSelected(config.getScreen().getEmulateGhosting());
148         enableGhosting.addActionListener(event -> {
149             config.getScreen().setEmulateGhosting(enableGhosting.isSelected());
150         });
151         
152         ghostingGroup.add(enableGhosting);
153         ghostingGroup.add(new JLabel());
154         
155         SpinnerModel fadeInModel = new SpinnerNumberModel(
156             config.getScreen().getGhosting().getAddBy(), 0.0, 1.0, 0.1);
157         fadeInModel.addChangeListener(event -> {
158             double addBy = (double) fadeInModel.getValue();
159             config.getScreen().getGhosting().setAddBy((float) addBy);
160         });
161         
162         SpinnerModel fadeOutModel = new SpinnerNumberModel(
163             config.getScreen().getGhosting().getSubtractBy(), 0.0, 1.0, 0.1);
164         fadeOutModel.addChangeListener(event -> {
165             double subBy = (double) fadeOutModel.getValue();
166             config.getScreen().getGhosting().setSubtractBy((float) subBy);
167         });
168         
169         JSpinner fadeIn = new JSpinner(fadeInModel);
170         JSpinner fadeOut = new JSpinner(fadeOutModel);
171         
172         ghostingGroup.add(new JLabel("Fade in"));
173         ghostingGroup.add(fadeIn);
174         ghostingGroup.add(new JLabel("Fade out"));
175         ghostingGroup.add(fadeOut);
176 
177         c.weighty = 0.5;
178         c.gridy = 2;
179         c.gridheight = 2;
180         
181         screenGroup.add(ghostingGroup, c);
182         
183         JPanel scaleGroup = new JPanel(new GridLayout(1, 2));
184         scaleGroup.add(new JLabel("Scale factor"));
185         SpinnerModel scaleModel = new SpinnerNumberModel(
186             config.getScreen().getDisplayScale(), 1, 50, 1);
187         scaleModel.addChangeListener(event -> {
188             int scale = (int) scaleModel.getValue();
189             config.getScreen().setScaleFactor(scale);
190         });
191         scaleGroup.add(new JSpinner(scaleModel));
192         
193         c.weighty = 0.8;
194         c.gridy = 4;
195         c.gridheight = 1;
196         
197         screenGroup.add(scaleGroup, c);
198         
199         c.weighty = 0.6f;
200         c.gridy = 0;
201         c.gridheight = 4;
202         
203         leftColumn.add(screenGroup, c);
204         
205         JPanel soundGroup = new JPanel(new GridLayout(1, 1));
206         soundGroup.setBorder(BorderFactory.createTitledBorder("Sound"));
207         JCheckBox enableBox = new JCheckBox("Enable?", config.getSound().isEnabled());
208         enableBox.addActionListener(event -> {
209             config.getSound().setEnabled(enableBox.isSelected());
210         });
211         soundGroup.add(enableBox);
212         
213         c.weighty = 0.4f;
214         c.gridy = 4;
215         c.gridheight = 1;
216         
217         leftColumn.add(soundGroup, c);
218         
219         c.weighty = 1.0f;
220         c.gridy = 5;
221         leftColumn.add(new JPanel(), c);
222         
223         add(leftColumn);
224         add(rightColumn);
225     }

227     public EmulatorConfiguration getConfig() {
228         return config;
229     }
230     
231     private void updateDisplayedBindings() {
232         Map<Integer, Integer> invertedBindings = MapUtils.invert(config.getInput().getBindings());
233 
234         for (int i = 0; i < keyBindBoxes.size(); i++) {
235             NamedKey key;
236 
237             if (invertedBindings.containsKey(i)) {
238                 key = NamedKey.fromCode(invertedBindings.get(i)).get();
239             } else {
240                 key = NamedKey.none();
241             }
242             
243             keyBindBoxes.get(i).setSelectedItem(key);
244         }
245     }
246     
247     private void updateConfigBindings() {
248         config.getInput().unbindAll();
249         
250         for (int i = 0; i < keyBindings.length; i++) {
251             if (keyBindings[i] != null) {
252                 config.getInput().bind(keyBindings[i], i);
253             } else {
254                 config.getInput().unbindDeviceKey(i);
255             }
256         }
257     }
258     
259     private void updateConfigBindings(EmulatorConfiguration.InputConfiguration config) {
260         Arrays.fill(keyBindings, null);
261         
262         Map<Integer, Integer> invertedBindings = MapUtils.invert(config.getBindings());
263         for (int i = 0; i < keyBindings.length; i++) {
264             if (invertedBindings.containsKey(i)) {
265                 keyBindings[i] = invertedBindings.get(i);
266             } else {
267                 keyBindings[i] = null;
268             }
269         }
270         
271         updateConfigBindings();
272     }
