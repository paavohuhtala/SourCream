2   package paavohuh.sourcream.ui;
3   
4   import java.awt.event.KeyEvent;
5   import java.util.ArrayList;
6   import java.util.List;
7   import java.util.Objects;
8   import java.util.Optional;
9   
10  /**
11   * A named key, consisting of a key code and a name.
12   */
13  public class NamedKey {
14      private final Integer keyCode;
15      private final String name;
16      private final boolean isNone;
17  
18      /**
19       * Creates a new named key.
20       * @param keyCode The Swing virtual keycode.
21       * @param name The human-friendly name of the key.
22       */
23      public NamedKey(int keyCode, String name) {
24          this.keyCode = keyCode;
25          this.name = name;
26          this.isNone = false;
27      }
28      
29      private NamedKey() {
30          this.isNone = true;
31          this.keyCode = null;
32          this.name = "None";
33      }
34      
35      /**
36       * Gets a list of all known named keys.
37       * @return A list of named keys.
38       */
39      public static List<NamedKey> getKnown() {
40          List<NamedKey> keys = new ArrayList<>();
41          
42          keys.add(none());
43          keys.add(new NamedKey(KeyEvent.VK_W, "W"));
44          keys.add(new NamedKey(KeyEvent.VK_A, "A"));
45          keys.add(new NamedKey(KeyEvent.VK_S, "S"));
46          keys.add(new NamedKey(KeyEvent.VK_D, "D"));
47          
48          keys.add(new NamedKey(KeyEvent.VK_UP, "Up"));
49          keys.add(new NamedKey(KeyEvent.VK_LEFT, "Left"));
50          keys.add(new NamedKey(KeyEvent.VK_DOWN, "Down"));
51          keys.add(new NamedKey(KeyEvent.VK_RIGHT, "Right"));
52          
53          keys.add(new NamedKey(KeyEvent.VK_Z, "Z"));
54          keys.add(new NamedKey(KeyEvent.VK_X, "X"));
55          keys.add(new NamedKey(KeyEvent.VK_SPACE, "Space"));
56          keys.add(new NamedKey(KeyEvent.VK_ENTER, "Enter"));
57          
58          return keys;
59      }
60      
61      /**
62       * Returns an unbound key. 
63       * @return An unbound key.
64       */
65      public static NamedKey none() {
66          return new NamedKey();
67      }
68      
69      /**
70       * Gets a NamedKey from a virtual key code.
71       * @param code The virtual key code.
72       * @return A NamedKey if it exists.
73       */
74      public static Optional<NamedKey> fromCode(int code) {
75          return getKnown().stream()
76              .filter(k -> k.keyCode != null && k.keyCode == code)
77              .findFirst();
78      }
79  
80      public Integer getKeyCode() {
81          return keyCode;
82      }
83  
84      public String getName() {
85          return name;
86      }
87  
88      public boolean isNone() {
89          return isNone;
90      }
91      
92      @Override
93      public String toString() {
94          return getName();
95      }
96  
97      @Override
98      public int hashCode() {
99          int hash = 7;
100         hash = 97 * hash + Objects.hashCode(this.keyCode);
101         hash = 97 * hash + Objects.hashCode(this.name);
102         hash = 97 * hash + (this.isNone ? 1 : 0);
103         return hash;
104     }
105 
106     @Override
107     public boolean equals(Object obj) {
108         if (obj == null) {
109             return false;
110         }
111         if (getClass() != obj.getClass()) {
112             return false;
113         }
114         final NamedKey other = (NamedKey) obj;
115         if (!Objects.equals(this.keyCode, other.keyCode)) {
116             return false;
117         }
118         if (!Objects.equals(this.name, other.name)) {
119             return false;
120         }
121         if (this.isNone != other.isNone) {
122             return false;
123         }
124         return true;
125     }
126 }
