2   package paavohuh.sourcream.configuration;
3   
4   import java.awt.event.KeyEvent;
5   import java.util.HashMap;
6   import static paavohuh.sourcream.configuration.EmulatorConfiguration.*;
7   
8   /**
9    * Contains premade input configurations for known applications and use cases.
10   */
11  public class KnownBindings {
12      /**
13       * Returns the premade input configuration for Tetris.
14       * Left and right arrow keys are used for movement. Up arrow is used for 
15       * rotating the tetromino. Down arrow is used for accelerating the fall.
16       * @return An input configuration.
17       */
18      public static InputConfiguration tetris() {
19          HashMap<Integer, Integer> bindings = new HashMap<>();
20          
21          bindings.put(KeyEvent.VK_LEFT, 5);
22          bindings.put(KeyEvent.VK_RIGHT, 6);
23          bindings.put(KeyEvent.VK_UP, 4);
24          bindings.put(KeyEvent.VK_DOWN, 7);
25          
26          return new InputConfiguration(bindings);
27      }
28      
29      /**
30       * Returns the premade input configuration for 2-player Pong.
31       * Player 1 uses W and S for movement, player 2 uses up and down arrow keys.
32       * @return An input configuration.
33       */
34      public static InputConfiguration pong() {
35          HashMap<Integer, Integer> bindings = new HashMap<>();
36          
37          bindings.put(KeyEvent.VK_W, 1);
38          bindings.put(KeyEvent.VK_S, 4);
39          bindings.put(KeyEvent.VK_UP, 12);
40          bindings.put(KeyEvent.VK_DOWN, 13);
41              
42          return new InputConfiguration(bindings);
43      }
44      
45      /**
46       * Returns an input configuration with every button bound to something.
47       * Keys 0 - 9 are bound to Q - P on a standard WASD keyboard. Keys 10 - 15
48       * are bound to A - H.
49       * This configuration is useful if you don't know which keys should be
50       * mapped.
51       * @return An input configuration.
52       */
53      public static InputConfiguration allBound() {
54          HashMap<Integer, Integer> bindings = new HashMap<>();
55          
56          bindings.put(KeyEvent.VK_Q, 0);
57          bindings.put(KeyEvent.VK_W, 1);
58          bindings.put(KeyEvent.VK_E, 2);
59          bindings.put(KeyEvent.VK_R, 3);
60          bindings.put(KeyEvent.VK_T, 4);
61          bindings.put(KeyEvent.VK_Y, 5);
62          bindings.put(KeyEvent.VK_U, 6);
63          bindings.put(KeyEvent.VK_I, 7);
64          bindings.put(KeyEvent.VK_O, 8);
65          bindings.put(KeyEvent.VK_P, 9);
66          bindings.put(KeyEvent.VK_A, 10);
67          bindings.put(KeyEvent.VK_S, 11);
68          bindings.put(KeyEvent.VK_D, 12);
69          bindings.put(KeyEvent.VK_F, 13);
70          bindings.put(KeyEvent.VK_G, 14);
71          bindings.put(KeyEvent.VK_H, 15);
72          
73          return new InputConfiguration(bindings);
74      }
