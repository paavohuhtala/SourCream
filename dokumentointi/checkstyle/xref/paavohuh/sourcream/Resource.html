2   package paavohuh.sourcream;
3   
4   import paavohuh.sourcream.emulation.ScreenBuffer;
5   import paavohuh.sourcream.emulation.ScreenBufferLoader;
6   import paavohuh.sourcream.utils.Cached;
7   import paavohuh.sourcream.utils.ResourceUtils;
8   
9   /**
10   * Contains methods for getting known built in resources (such as the logo or 
11   * the system font) from the .jar.
12   */
13  public class Resource {
14      /**
15       * The path to the SourCream logo.
16       */
17      public static final String LOGO = "/buffers/logo.buffer";
18      private static final Cached<ScreenBuffer> CACHED_LOGO;
19      
20      /**
21       * Gets the SourCream logo as a screen buffer.
22       * @return The screen buffer.
23       */
24      public static ScreenBuffer getLogo() {
25          return CACHED_LOGO.get();
26      }
27      
28      /**
29       * The path to the Chip-8 system font.
30       */
31      public static final String SYSTEM_FONT = "/systemfont.bin";
32      private static final Cached<byte[]> CACHED_SYSTEM_FONT;
33      
34      /**
35       * Gets the system font as a byte array.
36       * @return The system font.
37       */
38      public static byte[] getSystemFont() {
39          return CACHED_SYSTEM_FONT.get();
40      }
41      
42      /**
43       * Initializes the cached resources.
44       */
45      static {
46          CACHED_LOGO = new Cached<>(() ->
47              ResourceUtils.getAsString(LOGO).flatMap(ScreenBufferLoader::tryLoad).get());
48          
49          CACHED_SYSTEM_FONT = new Cached<>(() ->
50              ResourceUtils.getAsBytes(SYSTEM_FONT).get());
51      }
52  }
