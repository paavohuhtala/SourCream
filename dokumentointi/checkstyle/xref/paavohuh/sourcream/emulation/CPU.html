1   package paavohuh.sourcream.emulation;

3   import java.util.Optional;
4   import org.joou.UShort;
5   
6   /**
7    * The virtual CPU of the device. Contains the machine state, and handles the
8    * fetch-decode-execute -loop.
9    */
10  public class CPU {
11      private State state;
12      private InstructionDecoder decoder;
13      
14      public CPU(InstructionDecoder decoder, State state) {
15          this.state = state;
16      }
17      
18      /**
19       * Runs the CPU for one cycle.
20       * 1. Fetches an instruction
21       * 2. Decodes the instruction
22       * 3. Replaces current state with modified state
23       * @throws paavohuh.sourcream.emulation.UnknownInstructionException
24       */
25      public void runCycle() throws UnknownInstructionException {
26          UShort code = state.get16BitsAt(state.getProgramCounter());
27          Optional<Instruction> decoded = decoder.decode(code);
28          
29          if (!decoded.isPresent()) {
30              throw new UnknownInstructionException(code);
31          }
32          
33          Instruction instuction = decoded.get();
34          State newState = instuction.execute(state.withIncrementedPc());
35          
36          this.state = newState;
37      }
38  }
