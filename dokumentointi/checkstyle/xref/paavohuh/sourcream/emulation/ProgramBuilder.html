4   public class ProgramBuilder {
5       public static byte[] assemble(Instruction... instructions) {
6           byte[] byteCode = new byte[instructions.length * 2];
7           int i = 0;
8           
9           for (Instruction instr : instructions) {
10              int code = instr.getCode().intValue();
11              byteCode[i++] = (byte) ((code >>> 8) & 0xFF);
12              byteCode[i++] = (byte) (code & 0xFF);
13          }
14          
15          return byteCode;
16      }
17  }
