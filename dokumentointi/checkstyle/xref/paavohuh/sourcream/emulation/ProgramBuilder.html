4   public class ProgramBuilder {
5       /**
6        * Builds a new program from a list of instructions.
7        * @param instructions Variable number of instructions.
8        * @return A byte array containing the code for the program.
9        */
10      public static byte[] assemble(Instruction... instructions) {
11          byte[] byteCode = new byte[instructions.length * 2];
12          int i = 0;
13          
14          for (Instruction instr : instructions) {
15              int code = instr.getCode().intValue();
16              byteCode[i++] = (byte) ((code >>> 8) & 0xFF);
17              byteCode[i++] = (byte) (code & 0xFF);
18          }
19          
20          return byteCode;
21      }
