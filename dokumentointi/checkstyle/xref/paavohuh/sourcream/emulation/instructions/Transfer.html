4   import org.joou.UShort;

8   import paavohuh.sourcream.emulation.Register;

10  /**
11   * Contains transfer instructions.
12   * This class shouldn't be instantiated.
13   */
14  public class Transfer {
15      
16      /**
17       * Sets the address register (VI) to a constant. 
18       */
19      public static class SetAddressRegister extends Instruction.WithAddress {
20  
21          public SetAddressRegister(UShort address) {
22              super(address);
23          }
24  
25          @Override
26          public State execute(State state) {
27              return state.withAddressRegister(address);
28          }
29  
30          @Override
31          protected UShort getBaseCode() {
32              return UShort.valueOf(0xA000);
33          }
34      }
35      
36      /**
37       * Sets register VX to register VY.
38       */
39      public static class MoveRegister extends Instruction.WithTwoRegisters {
40  
41          public MoveRegister(Register x, Register y) {
42              super(x, y);
43          }
44  
45          @Override
46          protected UShort getBaseCode() {
47              return UShort.valueOf(0x8000);
48          }
49  
50          @Override
51          protected int getRegXOffset() {
52              return 2;
53          }
54  
55          @Override
56          protected int getRegYOffset() {
57              return 1;
58          }
59  
60          @Override
61          public State execute(State state) {
62              return state.withRegister(registerX, state.getRegister(registerY));
63          }    
64      }
65      
66      public static Seq<Instruction> getAll() {
67          return Seq.concat(
68              getAllInstances(SetAddressRegister::new),
69              getAllInstances(MoveRegister::new));
70      }
71  }
