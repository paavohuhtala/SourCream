12  /**
13   * Contains graphics instructions.
14   * This class shouldn't be instantiated.
15   */
16  public final class Graphics {
17      
18      /**
19       * Clears the screen.
20       */
21      public static class ClearScreen implements Instruction {
22  
23          @Override
24          public State execute(State state) {
25              return state.withScreenBuffer(state.getScreenBuffer().cleared());
26          }
27  
28          @Override
29          public UShort getCode() {
30              return UShort.valueOf(0x00E0);
31          }
32      }
33      
34      public static class DrawSprite extends Instruction.WithTwoRegistersAnd4BitConstant {
35  
36          public DrawSprite(Register x, Register y, UByte constant) {
37              super(x, y, constant);
38          }
39  
40          @Override
41          protected int getConstantOffset() {
42              return 0;
43          }
44  
45          @Override
46          protected int getRegXOffset() {
47              return 2;
48          }
49  
50          @Override
51          protected int getRegYOffset() {
52              return 1;
53          }
54  
55          @Override
56          protected UShort getBaseCode() {
57              return UShort.valueOf(0xD000);
58          }
59  
60          @Override
61          public State execute(State state) {
62              byte[] buffer = state.getMemoryFrom(state.getAddressRegister(), constant.intValue());
63              
64              ScreenBuffer sprite = new ScreenBuffer(buffer);
65              ScreenBuffer modifiedScreen = state.getScreenBuffer().blit(sprite, getRegisterX(state).intValue(), getRegisterY(state).intValue());
66              
67              return state.withScreenBuffer(modifiedScreen).withRegister(Register.VF, UByte.valueOf(modifiedScreen.flipped ? 1 : 0));
68          }
69      }
70      
71      public static Seq<Instruction> getAll() {
72          return Seq.concat(
73              Seq.of(new ClearScreen()),
74              getAllInstances(DrawSprite::new)
75          );
76      }
77  }
